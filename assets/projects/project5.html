# ros_bridge_connection.py (수정 후)
import asyncio
import websockets

class RosBridgeClient:
    async def connect(self, url: str):
        self._ws = await websockets.connect(url)
        asyncio.create_task(self._recv_loop())

    async def subscribe(self, topic: str, msg_type: str, callback):
        await self._send({"op": "subscribe", "topic": topic, "type": msg_type})
        self._subs[topic].append(callback)

    async def _recv_loop(self):
        async for raw in self._ws:
            msg = json.loads(raw)
            for cb in self._subs.get(msg.get("topic"), []):
                await cb(msg.get("msg", {}))

문제 2 — 카메라 페이지 이탈 후 CPU 누수

문제 현상

CCTV 페이지에서 4스트림을 열면 CPU가 약 31%까지 상승하고, 페이지를 닫아도 CPU 사용률이 유지되는 현상이 발생했습니다. 

10:42:47 | MEM: 124.5MB | CPU: 43.4s   ← CCTV 닫고도 CPU 감소 없음

원인 분석

camera_service.py의 stream_front()가 무한 루프로 동작하면서 HTTP 연결 종료 시에도 ROS 토픽 구독을 해제하지 않는 구조였습니다. 매 HTTP 요청마다 subscribe()만 호출하고 unsubscribe()는 없었기 때문에, 페이지를 열수록 구독이 누적되었습니다. 

# camera_service.py (수정 전)
async def stream_front(self):
    interval = 1.0 / self.fps
    while True:                          # 무한 루프
        frame = await asyncio.wait_for(...)
        yield (b'--frame\r\n...' + buffer + b'\r\n')
        await asyncio.sleep(interval)
        # HTTP 클라이언트 연결 종료 시 GeneratorExit 발생하지만
        # finally 블록이 없어 구독 해제 로직이 실행되지 않음

해결 — 레퍼런스 카운팅 기반 구독 수명 관리

# camera_service.py (수정 후)
async def stream_front(self):
    # 첫 소비자 진입 시 구독 시작
    if not self._front_subscribed and self._front_topic_name:
        await bridge.subscribe(topic_name, topic_type, self._on_front_image)
        self._front_subscribed = True

    self._front_consumers += 1   # 카운터 증가
    try:
        while True:
            frame = await asyncio.wait_for(self._front_queue.get(), ...)
            yield (b'--frame\r\n...' + buffer + b'\r\n')
    finally:
        # HTTP 연결 종료(GeneratorExit) 시 항상 실행
        self._front_consumers -= 1
        if self._front_consumers == 0:
            await bridge.unsubscribe(topic_name, self._on_front_image)
            self._front_subscribed = False  # ROS 토픽 구독 해제

문제 3 — MongoDB DuplicateKey 오류

백엔드 시작 시마다 robotId_1 인덱스에서 { robotId: null } 중복으로 인한 DuplicateKey 오류가 발생했습니다. MongoDB의 unique 인덱스는 기본적으로 null 값도 중복으로 취급하는데, 덤프 데이터에 robotId 필드가 없거나 null인 문서가 여러 개 존재했던 것이 원인이었습니다. 

# connection.py (수정 후) — sparse=True 추가
await db.robots.create_index("robotId", unique=True, sparse=True)
await db.robots.create_index("name",    unique=True, sparse=True)
# sparse=True: null 또는 필드 없는 문서는 인덱스에서 제외
#              → null 값 중복 허용, 기존 문서 18건 보존

기존에 생성된 non-sparse 인덱스를 스크립트로 제거 후 재생성하여, 서버 시작 시 인덱스 오류가 완전히 사라졌습니다.

문제 4 — robot_patrol 레이스 컨디션

Navigate/Homing으로 목표에 도달한 뒤 다시 Navigate 명령을 보내면, 로봇이 즉시 waiting 상태로 복귀하고 이동을 시작하지 않는 현상이 있었습니다. 

[robot_patrol]: 경로에 도달했습니다. Waiting service 호출하여 대기모드로 전환.
← 이후 새 명령을 보내도 수 초 내에 다시 waiting으로 복귀

robot_patrol.cpp의 follow_path()에서 경로 도달 후 global_path_received_ 플래그와 path_queue_를 초기화하지 않아, 다음 명령 도착 시 follow_current_path()가 빈 큐를 "이미 도착한 상태"로 오해하는 레이스 컨디션이었습니다.

3D 맵 뷰어 도입

기존 ECharts 기반 2D 맵에서 Three.js 3D 뷰어(MapViewer3D.vue)로 전환했습니다. Blender로 제작한 GLB 오피스 모델 위에 로봇 실시간 위치와 경로 노드를 실좌표로 렌더링합니다.

UTM 좌표를 Three.js 좌표계로 변환하는 함수를 작성하고, 실제 로봇 2대의 GPS UTM 측정값으로 원점(UTM_ORIGIN)을 역산하여 GLB 모델과 약 5m 이내 오차로 정렬했습니다. 

const UTM_ORIGIN_X = 304411.645  // c101.world GPS origin → UTM 역산값
const UTM_ORIGIN_Y = 3892836.76

function utmToScene(utmX, utmY) {
  return {
    x:  utmX - UTM_ORIGIN_X,   // UTM East  → Three.js +X
    z: -(utmY - UTM_ORIGIN_Y), // UTM North → Three.js -Z (Y-up 반전)
  }
}

기타 개선 사항

프론트엔드 인코딩 깨짐 해결

Vue 파일 25개가 EUC-KR(CP949)로 저장되어 있어 Vite 컴파일 오류가 발생했습니다. Python 스크립트에서 \uXXXX 유니코드 이스케이프를 사용해 순수 ASCII로 파일을 재작성함으로써 해결했습니다.

deep watcher 범위 최소화

RobotMap.vue의 watch(robots, ..., { deep: true })가 2초마다 들어오는 로봇 위치 SSE 업데이트에 반응해 매번 SSE를 재연결하면서 노드 선택이 초기화되는 버그가 있었습니다. watcher를 robots.map(r => r.seq).join(',') 형태로 변경하여 로봇 수가 바뀔 때만 재연결하도록 수정했습니다.

bcrypt 해시 필드 매핑 버그

auth_repository.py에서 DB 필드 hashedPassword를 불필요하게 hashed_password로 rename하는 코드가 있어 verify_password(password, None)가 항상 False를 반환하던 버그를 수정했습니다.
